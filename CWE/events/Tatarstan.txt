#Nationalist Agitation
country_event = { 
	id = 8120707
	title = "Nationalist Agitation"
	desc = EVTDESC8120707
	picture = "tar_protest"
  
	fire_only_once = yes

	trigger = {
		tag = TAR
		year = 1992
		has_global_flag = 1992_start_date_flag


		war = no
		TAR = { vassal_of = RUS }

		government = socialist_democracy
		ruling_party_ideology = socialist

	}

	mean_time_to_happen = { months = 1 }


	option = {
		name = "Tatarstan must stand strong!"
		any_pop = { 
			limit = { culture = tatar }
			ideology = { factor = 0.05 value = nationalist }
			militancy = 3
			consciousness = 3
		}
		any_pop = { 
			limit = { culture = russian }
			ideology = { factor = 0.10 value = big_tent }
			militancy = 3
			consciousness = 3
		}
	}
}
